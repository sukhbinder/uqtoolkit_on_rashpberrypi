50 double LogPosterior(Array1D<double>& m, void *info)
51 {
52  double pi=4.*atan(1.);
53 
54  postAux* pinfo = (postAux*) info;
55 
56  double logprior=0;
57 
58  for(int ic=0;ic<m.Length();ic++){
59 
60  if(!strcmp(pinfo->priortype(ic).c_str(),"uniform")){
61  double a=pinfo->priorparam1(ic);
62  double b=pinfo->priorparam2(ic);
63  if (a>=b)
64  throw Tantrum("Prior bounds are incorrect!");
65  logprior -= log(b-a);
66  }
67 
68  else if(!strcmp(pinfo->priortype(ic).c_str(),"normal")){
69  double mu=pinfo->priorparam1(ic);
70  double sig=pinfo->priorparam2(ic);
71 
72  logprior -= 0.5*log(2.*pi*sig*sig);
73  logprior -= 0.5*pow((m(ic)-mu)/sig,2.0);
74  }
75  else
76  throw Tantrum("Only unifrom or normal priors are implemented!");
77  }
78 
79 
80  Array2D<double> data=pinfo->data;
81  Array1D<double> modelparams=pinfo->modelparams;
82  // Set the parameter values for the forward run
83  int chaindim=m.XSize();
84  for(int ic=0;ic<chaindim;ic++)
85  modelparams(pinfo->chainParamInd(ic))=m(ic);
86 
87  // Posterior parameter
88  // Either Proportionaility constant between signal and noise for likelihood construction
89  // or standard deviation itself
90  double stdpar=pinfo->postparams(0);
91 
92 
93 
94 
95 
96  // Compare the model data with measurement data according to the noise model
97  // to get the likelihood.
98  double std1;
99  double std2;
100  if(!strcmp(pinfo->noisetype.c_str(),"const_stn")){
101  std1=stdpar*umodel;
102  std2=stdpar*vmodel;
103  }
104  else if(!strcmp(pinfo->noisetype.c_str(),"const_stdev")){
105  std1=stdpar;
106  std2=stdpar;
107  }
108  else
109  throw Tantrum("Noise type is not recognized!");
110 
111  double var1=std1*std1;
112  double var2=std2*std2;
113  double sum=logprior;

115 
116  for (int i=0;i<nTot;i++){
117  sum = sum - pow( model( data(i,0),modelparams ) - data(i,1),2.0)/(2.*var) - 0.5 * log(2.*pi*var);
118  }
