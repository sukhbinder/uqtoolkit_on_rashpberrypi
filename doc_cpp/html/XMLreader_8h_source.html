42 RefPtr<XMLElement> readXMLTree(const string inFileName)
43 {
44 
45 // Compute the name of the input and write-back XML file
46  unsigned long int xmlLPos = inFileName.find(".xml");
47  unsigned long int xmlUPos = inFileName.find(".XML");
48 
49  string outFileExt = ".parsed.xml";
50 
51  string outFileName = inFileName;
52  if(xmlLPos != string::npos){
53  outFileName.replace(xmlLPos,4,outFileExt);
54  }else if(xmlUPos != string::npos){
55  outFileName.replace(xmlUPos,4,outFileExt);
56  }else{
57  throw Tantrum(std::string("The XML input file ") + inFileName +
58  " does not have a valid (.xml or .XML) extension");
59  }
60 
61  ifstream xmlInFile(inFileName.c_str(),ios::in);
62  if(!xmlInFile){
63  throw Tantrum(std::string("Could not open XML input file ") + inFileName);
64  }
65  ofstream xmlOutFile(outFileName.c_str(),ios::out);
66  if(!xmlOutFile){
67  throw Tantrum(std::string("Could not open XML output file ") + outFileName);
68  }
69 
70  // Parse the input xml file
71  RefPtr<XMLParser> parser = new XMLExpatParser;
72  RefPtr<XMLElement> xmlTree = parser->parse(xmlInFile);
73 
74  // Write the parsed tree back to the write-back file
75  dump_xml_tree(xmlTree, "", xmlOutFile);
76 
77 return xmlTree;
78 
79 }

127 void readXMLUncInput(RefPtr<XMLElement> xmlTree, Array2D<double>& allPCcoefs,Array1D<int>& uncParamInd,int* outOrder, string* PCtype)
128 {
129 
130  RefPtr<XMLElement> pcInput = xmlTree->get_child("PC");
131  *PCtype = pcInput->attributes()->get("type","GH");
132  *outOrder = pcInput->attributes()->get_int("order",1);
133  string inpc_type= pcInput->attributes()->get("inpc","default");
134 
135  RefPtr<XMLElement> inpc_types = pcInput->get_child("inpc_types");
136  RefPtr<XMLElement> pt = inpc_types->get_child(inpc_type);
137 
138  bool noFileFlag=true;
139  string inpcfile;
140  if(!strcmp(inpc_type.c_str(),"file")) {
141  noFileFlag=false;
142  inpcfile=pt->attributes()->get("name","input_pc.dat");
143 
144  }
145 
146 
147  RefPtr<XMLElement> params = xmlTree->get_child("ModelParameters");
148  int np=params->count_children();
149 
150  uncParamInd.Clear();
151  Array1D<string> types(np);
152  // count the number of uncertain params
153  int ic=0;
154 
155  for (int i=0;i<np;i++){
156  types(i)=params->get_child(i)->attributes()->get("type","no_type_given");
157  if(!strcmp(types(i).c_str(),"uncertain")){
158  uncParamInd.PushBack(i);
159  ic++;
160  }
161  }
162  int uncdim=ic;
163  int npc=computeNPCTerms(uncdim,(*outOrder));
164  allPCcoefs.Resize(npc,np,0.e0);
165  for (int i=0;i<np;i++){
166 
167  double val=params->get_child(i)->attributes()->get_double("value",1.0);
168  allPCcoefs(0,i)=val;
169  }
170 
171  if(noFileFlag){
172 
173  // Reset the counter
174  ic=0;
175  for (int i=0;i<np;i++){
176 
177  if(!strcmp(types(i).c_str(),"uncertain")){
178  for(int ip=1;ip<=(*outOrder);ip++){
179  char buff[100];
180  sprintf(buff, "cf_%d", ip);
181  string cf_str = buff;
182  Array1D<int> mi(uncdim,0);
183  mi(ic)=ip;
184  allPCcoefs(get_invmindex(mi),i)=params->get_child(i)->attributes()->get_double(cf_str,0.0);
185  }
186 
187 
188 
189  ic++;
190  }
191 
192 }
193  }
194 
195  else{
196  Array2D<double> uncPCcoefs(npc,uncdim,0.e0);
197  read_datafile(uncPCcoefs,inpcfile.c_str());
198  for(int ic=0;ic<uncdim;ic++)
199  for(int j=0;j<npc;j++)
200  allPCcoefs(j,uncParamInd(ic))=uncPCcoefs(j,ic);
201  }
202 
203 
204 
205  return;
206 }

226 void readXMLChainInput(RefPtr<XMLElement> xmlTree,MCMC* pmchain, Array1D<double>& chstart, int* nsteps, Array1D<int>& chainParamInd, Array1D<string>& priortype, Array1D<double>& priorparam1, Array1D<double>& priorparam2)
227 {
228 
229  // Select the tree element with the input parameters for this run
230  RefPtr<XMLElement> mcmcInput = xmlTree->get_child("MCMC");
231  string name = mcmcInput->attributes()->get("name","no_name_specified");
232  string method = mcmcInput->attributes()->get("method","no_method_specified");
233  string chainparam = mcmcInput->attributes()->get("chainparam","no_chainparam_specified");
234  string output = mcmcInput->attributes()->get("output","no_output_specified");
235  cout << "Name : " << name << endl;
236  cout << "Method : " << method << endl;
237  cout << "Chainparam : " << chainparam << endl;
238  cout << "Output : " << output << endl;
239 
240  RefPtr<XMLElement> method_types = mcmcInput->get_child("method_types");
241  RefPtr<XMLElement> pt1=method_types->get_child(method);
242  double gamma=pt1->attributes()->get_double("gamma",1.);
243  double eps_cov=pt1->attributes()->get_double("eps_cov",1e-8);
244 
245  int adaptstart=pt1->attributes()->get_int("adstart",1000);
246  int adaptstep=pt1->attributes()->get_int("adstep",10);
247  int adaptend=pt1->attributes()->get_int("adstop",1000000);
248  (*nsteps)=pt1->attributes()->get_int("nsteps",10000);
249 
250 
251  RefPtr<XMLElement> chainparam_types = mcmcInput->get_child("chainparam_types");
252  RefPtr<XMLElement> pt2=chainparam_types->get_child(chainparam);
253 
254  Array1D<double> chsig;
255  bool noFileFlag;
256 
257  if(!strcmp(chainparam.c_str(),"file")) {
258  noFileFlag=false;
260  read_datafileVS(chstart, pt2->attributes()->get("chstart","chain_start.dat").c_str());
261  read_datafileVS(chsig, pt2->attributes()->get("chsig","chain_sig.dat").c_str());
262  }
263  else
264  noFileFlag=true;
265 
266  RefPtr<XMLElement> params = xmlTree->get_child("ModelParameters");
267  int np=params->count_children();
268 
269  chainParamInd.Clear();
270  priortype.Clear();
271  priorparam1.Clear();
272  priorparam2.Clear();
273 
274  for (int i=0;i<np;i++){
275 
276  string type=params->get_child(i)->attributes()->get("type","no_type_given");
277  if(!strcmp(type.c_str(),"infer")){
278  chainParamInd.PushBack(i);
279  if(noFileFlag){
280  double val=params->get_child(i)->attributes()->get_double("value",1.0);
281  chstart.PushBack(val);
282  chsig.PushBack(params->get_child(i)->attributes()->get_double("sigma",MAX(fabs(0.01*val),0.001)));
283  }
284 
285  priortype.PushBack(params->get_child(i)->attributes()->get("prior","uniform"));
286  priorparam1.PushBack(params->get_child(i)->attributes()->get_double("pr1",0.0));
287  priorparam2.PushBack(params->get_child(i)->attributes()->get_double("pr2",1000.0));
288  }
289  }
290 
291  int chdim=chainParamInd.XSize();
292 
293 
294 
295  RefPtr<XMLElement> output_types = mcmcInput->get_child("output_types");
296  RefPtr<XMLElement> pt3=output_types->get_child(output);
297  string filename=pt3->attributes()->get("file","chain.dat");
298  int freq_outscreen=pt3->attributes()->get_int("screen",1000);
299  int freq_chainfile=pt3->attributes()->get_int("freq",1000);
300 
301  pmchain->setChainDim(chdim);
302  pmchain->initMethod(method);
303 
304  // \todo Need to set checks since the next three lines only useful for method=am
305  pmchain->initAdaptSteps(adaptstart,adaptstep,adaptend);
306  pmchain->initAMGamma(gamma);
307  pmchain->initEpsCov(eps_cov);
308 
309  pmchain->setOutputInfo(output,filename,freq_chainfile, freq_outscreen);
310  pmchain->initChainPropCovDiag(chsig);
311 
312 
313  return;
314 }
