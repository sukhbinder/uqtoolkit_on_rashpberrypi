162 void forwardFunctionDt(double* inpParams, double dTym, double* outValues)
163 {
164  // variables to store right hand side
165  double dudt = 0.e0;
166  double dvdt = 0.e0;
167  double dwdt = 0.e0;
168 
169  // parse input parameters
170  const double a=inpParams[0];
171  const double b=inpParams[1];
172  const double c=inpParams[2];
173  const double d=inpParams[3];
174  const double e=inpParams[4];
175  const double f=inpParams[5];
176 
177  // Save solution at current time step
178  double u_o = outValues[0];
179  double v_o = outValues[1];
180  double w_o = outValues[2];
181  double z_o = 1.0 - u_o - v_o - w_o;
182 
183  // Integrate with 2nd order Runge Kutta
184 
185  // Compute right hand sides
186  GetRHS(a,b,c,d,e,f,u_o,v_o,w_o,z_o,dudt,dvdt,dwdt);
187 
188  // Advance to mid-point
189  double u = u_o + 0.5*dTym*dudt;
190  double v = v_o + 0.5*dTym*dvdt;
191  double w = w_o + 0.5*dTym*dwdt;
192  double z = 1.0 - u - v - w;
193 
194  // Compute right hand sides
195  GetRHS(a,b,c,d,e,f,u,v,w,z,dudt,dvdt,dwdt);
196 
197  // Advance to next time step
198  u = u_o + dTym*dudt;
199  v = v_o + dTym*dvdt;
200  w = w_o + dTym*dwdt;
201  z = 1.0 - u - v - w;
202 
203  // Store the output
204  outValues[0]=u;
205  outValues[1]=v;
206  outValues[2]=w;
207 
208  return;
209 }

225 void GetRHS(const PCSet& aPCSet, const double* a, const double* b, const double* c, const double* d, const double* e, const double* f,
226  const double* u, const double* v, const double* w, const double* z,
227  double* dudt, double* dvdt, double* dwdt)
228 {
229  // Number of PC terms
230  const int nPCTerms = aPCSet.GetNumberPCTerms();
231 
232  // Work variable
233  double* dummy1 = new double[nPCTerms];
234  double* dummy2 = new double[nPCTerms];
235  double* dummy3 = new double[nPCTerms];
236 
237  // Build du/dt = a*z - c*u - 4.0*d*u*v
238  aPCSet.Prod(z,a,dummy1); // dummy1 = a*z
239  aPCSet.Prod(u,c,dummy2); // dummy2 = c*u
240  aPCSet.SubtractInPlace(dummy1,dummy2); // dummy1 = a*z - c*u
241  aPCSet.Prod(u,v,dummy2); // dummy2 = u*v
242  aPCSet.Prod(dummy2,d,dummy3); // dummy3 = d*u*v
243  aPCSet.MultiplyInPlace(dummy3,4.e0); // dummy3 = 4.0*d*u*v
244  aPCSet.Subtract(dummy1,dummy3,dudt); // dudt = a*z - c*u - 4.0*d*u*v
245 
246  // Build dv/dt = 2.0*b*z*z - 4.0*d*u*v
247  aPCSet.Prod(z,z,dummy1); // dummy1 = z*z
248  aPCSet.Prod(dummy1,b,dummy2); // dummy2 = b*z*z
249  aPCSet.Multiply(dummy2,2.e0,dummy1); // dummy1 = 2.0*b*z*z
250  aPCSet.Prod(u,v,dummy2); // dummy2 = u*v
251  aPCSet.Prod(dummy2,d,dummy3); // dummy3 = d*u*v
252  aPCSet.MultiplyInPlace(dummy3,4.e0); // dummy3 = 4.0*d*u*v
253  aPCSet.Subtract(dummy1,dummy3,dvdt); // dvdt = 2.0*b*z*z - 4.0*d*u*v
254 
255  // Build dw/dt = e*z - f*w
256  aPCSet.Prod(z,e,dummy1); // dummy1 = e*z
257  aPCSet.Prod(w,f,dummy2); // dummy2 = f*w
258  aPCSet.Subtract(dummy1,dummy2,dwdt); // dwdt = e*z - f*w
259 
260  // Memory clean-up
261  delete [] dummy1;
262  delete [] dummy2;
263  delete [] dummy3;
264 
265  return;
266 }
